from datetime import datetime

def _extract_freq_amp_arrays(analysis_obj):



    data = getattr(analysis_obj, "data", {}) or {}



    freq_candidates = ["fqs_MHz", "freqs_MHz", "rf_MHz", "x_MHz", "fqs", "freqs", "rf"]



    amp_candidates = ["mags", "magnitude", "amp", "y", "response"]







    fq = None



    for k in freq_candidates:



        if k in data:



            fq = np.asarray(data[k], dtype=float)



            break



    amp = None



    for k in amp_candidates:



        if k in data:



            amp = np.asarray(data[k], dtype=float)



            break



    if fq is not None and np.nanmax(np.abs(fq)) > 1e8:



        fq = fq / 1e6



    return fq, amp







attempts = [



    dict(rf_begin=6140.0, rf_end=6160.0, df_khz=100, qb_gain=0.04, qb_len=20000, n_avg=4000),



    dict(rf_begin=6120.0, rf_end=6180.0, df_khz=100, qb_gain=0.05, qb_len=24000, n_avg=4000),



]







pre_qb_fq = float(getattr(attr, "qb_fq", np.nan))



best = {"status": "failed", "f01_mhz": np.nan, "method": "none", "attempt": None, "apply": None}







for i, a in enumerate(attempts, start=1):

            "qb_gain": a["qb_gain"],



            "qb_len": a["qb_len"],

            "n_avg": a["n_avg"],

        apply=False,



        persist_artifact=True,



    )



    qb_result_try = orch._run_result_from_artifact(qb_cycle_try["artifact"])



    qb_analysis_try = qb_spec.analyze(qb_result_try, update_calibration=False)



    qb_spec.plot(qb_analysis_try)







    f0_mhz = float(qb_cycle_try["calibration_result"].params.get("f0_MHz", np.nan))
